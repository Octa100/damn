# Implement Named Entity Recognition (NER)

import nltk
nltk.download('averaged_perceptron_tagger')
nltk.download('maxent_ne_chunker')
nltk.download('words')
nltk.download('punkt')

from nltk import ne_chunk

m = "The US President stays in white house"
nt=nltk.word_tokenize(m)
ntg=nltk.pos_tag(nt)
nr=ne_chunk(ntg)

print(nr)
