# Implement syntactic parsing of a given text

import nltk

nltk.download('punkt')
nltk.download('averaged_perceptron_tagger_eng')

text = "The big cat ate the little mouse who was after fresh cheese"

tokens = nltk.word_tokenize(text)
tagged = nltk.pos_tag(tokens)

print("POS Tags:")
print(tagged)

grammar_np = r"NP: {<DT>?<JJ>*<NN>}"

chunk_parser = nltk.RegexpParser(grammar_np)
chunk_result = chunk_parser.parse(tagged)

print("\nChunk Tree:")
print(chunk_result)

# Opens a tree window
chunk_result.draw()
