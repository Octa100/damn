# Implement stemming and lemmatization

import nltk
from nltk.stem import PorterStemmer, WordNetLemmatizer
from nltk.tokenize import word_tokenize

nltk.download('punkt')
nltk.download('wordnet')

ps = PorterStemmer()
for w in ["programs","programer","programing","programmers"]:
    print(w, ":", ps.stem(w))

for w in word_tokenize("programmers program with programing languages"):
    print(w, ":", ps.stem(w))

lemma = WordNetLemmatizer()
for w in ["rocks","corpora","better","best","worst"]:
    print(w, ":", lemma.lemmatize(w))

print("better:", lemma.lemmatize("better", pos="a"))
